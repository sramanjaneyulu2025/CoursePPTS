CREATE TABLE inventory(inventory_id INT PRIMARY KEY,
                       product_id INT,
                       quantity INT NOT NULL,
                       warehouse_id INT,
                       last_updated TIMESTAMP,
                       FOREIGN KEY(product_id) REFERENCES products(product_id));
